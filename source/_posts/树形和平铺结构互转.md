# 1. 将树形转换成平铺数据结构


```js
  const entry = {
            a: {
                b: {
                    c: {
                        dd: 'abcdd'
                    }
                },
                d: {
                    ee: 'adee'
                },
                f: 'af'
            }
        }
        const output = {
            'a.b.c.dd': 'abcdd',
            'a.d.ee': 'adee',
            'a.f': 'af',
        }
```

## 解答一：递归 

```js
function flat(obj,preKey='',res={}){
    for(const key in obj){
        const newKey=`${preKey}${key}`
        if(typeof obj[key] ==='object'){
            flat(obj[key],`${newKey}.`,res)
        }else{
            res[newKey]=obj[key]
        }
    }
    return res
}
```
